\resumeSubheadingLight{Mechanization of Correctness Proofs for Concurrent Objects in $\mathsf{L\bm{\exists\forall}N}$}{}
\resumeItemListStartNoInd
\resumeItem{
  Ongoing work exploring the mechanization and automation of safety proofs for lock-free concurrent objects.}
\resumeItem{
  Developing a compositional proof framework that derives trace quotients by combining partial-order reduction and Lipton's reduction via induction-reduction tuples capturing recurrent nonblocking synchronization patterns.}
\resumeItem{
  Demonstrating expressiveness by verifying canonical lock-free data structures, including a concurrent counter, Treiber Stack, Michael-Scott Queue, Elimination Stack.}

\resumeSubheadingLight{$\bm{\mathsf{ev}}\mathsf{Drift}$: Static Analysis of Dependent Temporal Safety Effects of Functional Programs}{}
\resumeItemListStartNoInd
\resumeItem{Developed a scalable static analysis for automated verification of temporal safety properties over effect traces for a subset of OCaml, equipped with an $\mathsf{ev}$ expression that emits events as side effects}
\resumeItem{Built an automata-based abstract interpreter for the trace extension operator $\mathsf{ev}$, integrating it into the abstract-interpretation-based dataflow refinement type inference tool $\mathsf{Drift}$}
\resumeItem{Demonstrated improved scalability and performance compared to state-of-the-art automated safety verifiers}
\resumeItemListEnd
